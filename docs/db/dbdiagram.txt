table users {
  id bigint [pk]

  name varchar [not null]
  email varchar [not null, unique]
  password varchar [not null]

  role user_role [not null, default: 'officer']

  created_at timestamp [not null]
  updated_at timestamp [not null]
  deleted_at timestamp [null]

  indexes {
    email [unique]
    role
  }
}

table inmates {
  id bigint [pk]

  registration_number varchar [not null, unique]
  name varchar [not null]

  gender gender_type [null]

  crime_type varchar [null]

  admission_date date [null]
  placement_date date [null]
  expiration_date date [null]

  status inmate_status [not null, default: 'active']

  current_room_id bigint [null]

  created_at timestamp [not null]
  updated_at timestamp [not null]
  deleted_at timestamp [null]

  indexes {
    registration_number [unique]
    name
    current_room_id
    status
    admission_date
    expiration_date
  }
}

Ref: inmates.current_room_id > rooms.id [
  update: cascade,
  delete: set null
]

table blocks {
  id bigint [pk]

  code varchar [not null, unique]
  name varchar [not null]

  status block_status [not null, default: 'active']

  created_at timestamp [not null]
  updated_at timestamp [not null]
  deleted_at timestamp [null]

  indexes {
    code [unique]
    status
  }
}

table rooms {
  id bigint [pk]

  block_id bigint [not null]

  code varchar [not null, unique]
  name varchar [null]

  capacity int [not null, default: 0]
  current_occupancy int [not null, default: 0]

  status room_status [not null, default: 'active']

  created_at timestamp [not null]
  updated_at timestamp [not null]
  deleted_at timestamp [null]

  Note: 'current_occupancy <= capacity'

  indexes {
    block_id
    code [unique]
    status
    current_occupancy
  }
}

Ref: rooms.block_id > blocks.id [
  update: cascade,
  delete: restrict
]

table room_transfers {
  id bigint [pk]

  transfer_number varchar [not null, unique]

  inmate_id bigint [not null]

  room_from_id bigint [not null]
  room_to_id bigint [not null]

  reason transfer_reason_type [null]

  notes text [null]

  transferred_at datetime [not null]

  status transfer_status [not null, default: 'completed']

  officer_name varchar [not null]
  officer_signature longtext [null]

  created_by bigint [not null]

  created_at timestamp [not null]
  updated_at timestamp [not null]
  deleted_at timestamp [null]

  indexes {
    transfer_number [unique]
    inmate_id
    room_from_id
    room_to_id
    created_by
    transferred_at
    status
    (inmate_id, transferred_at)
    (room_to_id, transferred_at)
  }
}

Ref: room_transfers.inmate_id > inmates.id [
  update: cascade,
  delete: restrict
]

Ref: room_transfers.room_from_id > rooms.id [
  update: cascade,
  delete: restrict
]

Ref: room_transfers.room_to_id > rooms.id [
  update: cascade,
  delete: restrict
]

Ref: room_transfers.created_by > users.id [
  update: cascade,
  delete: restrict
]

enum user_role {
  admin
  officer
}

enum gender_type {
  male
  female
}

enum inmate_status {
  active
  released
  transferred
  deceased
}

enum block_status {
  active
  inactive
}

enum room_status {
  active
  inactive
  maintenance
}

enum transfer_status {
  completed
  cancelled
}

enum transfer_reason_type {
  disciplinary
  medical
  security
  request
  administration
  other
}
